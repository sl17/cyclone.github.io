### JS数组根据一定规则排序




### 一个数组根据另一个规则进行重新排序


```
let arr = [
	{ value: 'asdasda', id: '1' },
	{ value: 'asdasda', id: '2' },
	{ value: 'asdasda', id: '3' },
	{ value: 'asdasda', id: '4' },
	{ value: 'asdasda', id: '5' },
	{ value: 'asdasda', id: '6' },
	{ value: 'asdasda', id: '7' },
	{ value: 'asdasda', id: '8' },
	{ value: 'asdasda', id: '9' },
	{ value: 'asdasda', id: '10' },
	{ value: 'asdasda', id: '11' },
	{ value: 'asdasda', id: '12' },
]
console.log(JSON.parse(JSON.stringify(arr)));
let list = ['1', '3', '2', '4', '6', '7', '9', '12', '11', '10', '5', '8']
arr = arr.sort(function (a, b) {
	return (
		list.indexOf(a.id) -
		list.indexOf(b.id)
	)
})
console.log(arr);
```
