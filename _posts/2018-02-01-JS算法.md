//方法一
function sort(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = temp
      }
   }
 }
 return arr
}

去重
```
//方法一
function noRepeat(arr){
  let newArr = [];
  for(let i = 0;i < arr.length;i++){
    if(newArr.indexOf(arr[i]) === -1){
    	newArr.push(arr[i])
    }
  }
  return newArr
}

//方法二 利用ES6 Set() 去重

function noRepeat(arr){
  return [...new Set(arr)]}
}
```

function prime(){
  let arr = []
  for(let i = 2;i < 100;i++){
    let flag = true;
    for(let j = 2;j < Math.sqrt(i);j++){
      if(i % j == 0){
        flag = false;
        break
      }
    }
    if(flag){
      arr.push(i)
    }
  }
  return arr
}
