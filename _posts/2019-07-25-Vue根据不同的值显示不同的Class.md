### Vue根据传入不同的值显示不同的Class

```
style:

html,body{background: #eee;}
p{width: 200px;height: 50px;position: relative;margin-bottom: 10px;border: 1px solid #A71D5D;background: #fff;}
p:after{content: "";width: 100%;height: 12px;position: absolute;bottom: 0;left: 0;}
p.r:after{background: red;}/*100-200*/
p.b:after{background: deepskyblue;}/*200-300*/
p.y:after{background: yellow;}/*300-400*/
p.g:after{background: greenyellow;}/*400-500*/
p.o:after{background: orange;}/*500-600*/

html:

<div id="app">
 <p 
  v-for="item in arr" 
  :key="item.id" 
  :class="[{'r': item < 200 && item > 0},{'b': item < 300 && item > 199},{'y': item < 400 && item > 299},{'g': item < 500 && item > 399},{'o': item < 600 && item > 499}]"
 >{{item}}</p>
</div>

script:

new Vue({
 el: '#app',
 data:{
  msg: 'Vue',
  arr:[10,400,199,200,150,280,380,405,504,566,120],
 }
})

```
