## JS封装get请求传参拼接.


```
function decorateUrl(url, getBody = {}) {
    const data = [url]
    const array = Object.keys(getBody).map(v => `${v}=${getBody[v]}`);
    array.length > 0 && data.push(array.join("&"));
    return data.join('?');
}
let parmas = {
    id: 1313,
    // name: 'jdskfjd',
    // txt: 454
}
console.log(decorateUrl('http://baidu.com',parmas))
```
