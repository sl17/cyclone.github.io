### vue3子组件暴露方法

### 父组件

```
<child ref='child'></child>
<script setup lang="ts">
  const child = ref()
  child.value.showFn()
  // 或者
  child.value.getList()
</script>
```
### 子组件

```
<script setup lang="ts">
defineExpose({
  showFn () {
    getList()
  },
  // 或者
  getList
})
</script>

```
