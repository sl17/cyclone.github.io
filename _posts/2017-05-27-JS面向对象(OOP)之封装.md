## JS面向对象(OOP)之封装



```
function Person(name, age, sex){
	this.name = name;
	this.age = age;
	this.sex = sex;
	this.type = '人类';
	this.love = 'play';
}

Person.prototype.hobby = function(){
	alert(this.love)
}

var wang = new Person("王",20,'男')
console.log(wang.age) // 20
wang.hobby() //play

console.log(wang)//Person{name: "王", age: 20, sex: "男", type: "人类", love: "play"}
```
