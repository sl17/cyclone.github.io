## JS匹配两个数组id相同项过滤

```
let arr1 = [1, 2]
let arr2 = [{ id: 1, name: '老大' }, { id: 3, name: '老三' }, { id: 4, name: '老四' }, { id: 5, name: '老五' },]

let add = arr2.filter(item => {
  let t = arr1.map(v=>v)
  return !t.includes(item.id)
})
console.log(add)
//[{"id":3,"name":"老三"},{"id":4,"name":"老四"},{"id":5,"name":"老五"}]
```
