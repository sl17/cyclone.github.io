## Vue页面内部切换配置路由

```
<Menu mode="horizontal" @on-select="douStatus" :active-name="menu">
  <Menu-item name="0">项目1</Menu-item>
  <Menu-item name="1">项目2</Menu-item>
  <Menu-item name="2">项目3</Menu-item>
</Menu>
```

```
computed: {
  menu() {
    return this.$route.query.menu || "0";
  },
},
methods: {
  douStatus(menu) {
    this.$router.push({
      query: {
        menu
      }
    });
    if(menu === 0){
    	console.log("zhixing ")
    }
  },
},
```
