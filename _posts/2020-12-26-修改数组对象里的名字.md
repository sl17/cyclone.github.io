## 修改数组对象里的名字

let obj = [
  { name: 'Bob', age: 20, oo: 89564 }, 
  { name: 'Bob2', age: 21, oo: 8989 },
  { name: 'Bob3', age: 21, oo: 8989 },
  { name: 'Bob4', age: 21, oo: 8989 },
  { name: 'Bob5', age: 21, oo: 8989 },
]
let obj2 = obj.map(e=>({
  change_name: e.name,
  change_age: e.age,
  change_oo: e.oo,
}))
console.log(obj,obj2)
