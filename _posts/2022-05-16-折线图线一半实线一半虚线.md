## 折线图线一半实线一半虚线


Echart 折线图线一半实线一半虚线

> 注意 折线图的data项是一个个数组，不是对象

```
option = {
    tooltip: {
        trigger: 'axis'
    },
    legend: {
        data: ['邮件营销', '预测']
    },
  
    xAxis: {
        type: 'category',
        data: ['周一', '周二', '周三', '周四', '周五', '周六', '周日']
    },
    yAxis: {
        type: 'value'
    },
    series: [
        {
            name: '邮件营销',
            type: 'line',
            // 注意这里是一个个数组
            data: [['周一',120], ['周二',132],['周三',101],['周四',134]]
        },
         {
            name: '预测',
            type: 'line',
             lineStyle: {
                type: 'dashed',
            },
            data:[['周四',134],['周五',90], ['周六',230],['周日',210]]
        },
        {
            name: '联盟广告',
            type: 'line',
            data: [
                {name:'周一',value:220},
                {name:'周二',value:182}, 
                {name:'周三',value:191}, 
                {name:'周四',value:234},  
                {name:'周五',value:290},  
                {name:'周六',value:330}, 
                {name:'周日',value:310}
            ]
        },
      
    ]
};

```
