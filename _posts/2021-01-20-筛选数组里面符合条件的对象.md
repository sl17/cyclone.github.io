## 筛选数组里面符合条件的对象

```
let oldArr = [
  { id: 1, name: '11111' },
  { id: 2, name: '22222' },
  { id: 3, name: '33333' },
  { id: 4, name: '44444' },
  { id: 5, name: '55555' },
  { id: 6, name: '66666' },
]
let arr = [1, 2, 3]
oldArr = oldArr.filter((item) => {
  let t = arr.map((v) => v);
  return !t.includes(item.id);
});
console.log(oldArr)

//  { id: 4, name: '44444' },
//  { id: 5, name: '55555' },
//  { id: 6, name: '66666' },
//]

```
