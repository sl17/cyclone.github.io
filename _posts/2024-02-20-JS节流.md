## JS节流


### JS中节流函数的使用

```
<body>
	<input type="text">
	<script>
		const input1 = document.querySelector('input');

		const inputChange = function(e){
			console.log('ssss',this);
		}

		input1.oninput = debounce(inputChange,3000)

		function debounce(fn, delay) {
			let timer = null;
			return function () {
				let arg = arguments;
				clearTimeout(timer);
				timer = setTimeout(() => {
					fn.apply(this, arg)
				}, delay)
			}
		}

	</script>
</body>
```
