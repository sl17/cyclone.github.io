## 关于百度地图引入页面

```
<script type="text/javascript" src="http://api.map.baidu.com/getscript?v=2.0&amp;ak=r2ecxkl0CrFQ3BunjQ2mb5C3fj3gFRCG&amp;services=&amp;t=20170608143204"></script>

<script type="text/javascript">
jQuery(document).ready(function($) {
	var myIcon = new BMap.Icon("http://www.zjcheshi.com/images/map/friendicon.png", new BMap.Size(30,34)); // 自定义的标注图片  
	var map = new BMap.Map("dituContent"); //地图放在id为“dituContent”的div中
	var point = new BMap.Point(120.158318,30.310555); //定位坐标
	
	map.centerAndZoom(point, 15); //地图显示范围比例  （地图缩放到什么程度）
	var marker = new BMap.Marker(point,{icon:myIcon});  // 创建标注
	map.addOverlay(marker);               // 将标注添加到地图中
	marker.setAnimation(BMAP_ANIMATION_BOUNCE); //跳动的动画
})

</script>
```
